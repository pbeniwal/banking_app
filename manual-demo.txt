from flask import Flask, render_template, request, redirect, url_for, flash
from flask_mysqldb import MySQL

app = Flask(__name__)
app.secret_key = 'supersecretkey'

# MySQL Configuration
app.config['MYSQL_HOST'] = 'localhost'
app.config['MYSQL_USER'] = 'bank_user'
app.config['MYSQL_PASSWORD'] = 'your_password'
app.config['MYSQL_DB'] = 'banking_db'

mysql = MySQL(app)

@app.route('/')
def index():
    cur = mysql.connection.cursor()
    cur.execute("SELECT * FROM customers")
    customers = cur.fetchall()
    cur.close()
    return render_template('index.html', customers=customers)

@app.route('/create', methods=['POST'])
def create():
    name = request.form['name']
    acc_num = request.form['acc_num']
    cur = mysql.connection.cursor()
    try:
        cur.execute("INSERT INTO customers (name, account_number, balance) VALUES (%s, %s, 0)", (name, acc_num))
        mysql.connection.commit()
        flash("Customer Created Successfully!")
    except:
        flash("Error: Account number might already exist.")
    cur.close()
    return redirect(url_for('index'))

@app.route('/transaction', methods=['POST'])
def transaction():
    acc_num = request.form['acc_num']
    amount = float(request.form['amount'])
    action = request.form['action']
    
    cur = mysql.connection.cursor()
    cur.execute("SELECT balance FROM customers WHERE account_number = %s", [acc_num])
    result = cur.fetchone()

    if result:
        current_balance = float(result[0])
        if action == 'deposit':
            new_balance = current_balance + amount
        else: # withdrawal
            if current_balance >= amount:
                new_balance = current_balance - amount
            else:
                flash("Insufficient Funds!")
                return redirect(url_for('index'))
        
        cur.execute("UPDATE customers SET balance = %s WHERE account_number = %s", (new_balance, acc_num))
        mysql.connection.commit()
        flash(f"{action.capitalize()} Successful!")
    else:
        flash("Account Not Found!")
    
    cur.close()
    return redirect(url_for('index'))

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
